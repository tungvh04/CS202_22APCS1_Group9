435void World::handleCollisions() {
436 std::set<SceneNode*> playerCollidingNodes;
437 mSceneGraph.checkNodeCollision(mPlayerCharacter->getBoundingRect(), playerCollidingNodes);
438 for (auto node : playerCollidingNodes) {
439 if (matchesCategories(node, Category::Obstacle)) {
440 Command command;
441 command.category = Category::PlayerCharacter;
442 command.action = derivedAction<Character>([](Character& c, sf::Time) { c.destroy(); });
443 mCommandQueue.push(command);
444 }
445 if (matchesCategories(node, Category::Hot)) {
446 mPlayerCharacter->shiftTemperature(Constants::HotTemperatureShift);
447 }
448 if (matchesCategories(node, Category::Cold)) {
449 mPlayerCharacter->shiftTemperature(Constants::ColdTemperatureShift);
450 }
451 if (matchesCategories(node, Category::SpeedUp)) {
452 mPlayerCharacter->multSpeedMult(Constants::SpeedUpMult);
453 }
454 if (matchesCategories(node, Category::SlowDown)) {
455 mPlayerCharacter->multSpeedMult(Constants::SlowDownMult);
456 }
457 if (matchesCategories(node, Category::HealSmall)) {
458 mPlayerCharacter->hurt(-Constants::healAmountSmall);
459 }
460 if (matchesCategories(node, Category::HealLarge)) {
461 mPlayerCharacter->hurt(-Constants::healAmountLarge);
462 }
463 if (matchesCategories(node, Category::HurtSmall)) {
464 mPlayerCharacter->hurt(Constants::hurtAmountSmall);
465 }
466 if (matchesCategories(node, Category::HurtLarge)) {
467 mPlayerCharacter->hurt(Constants::hurtAmountLarge);
468 }
469 if (matchesCategories(node, Category::PickUp)) {
470 node->setDestroy();
471 }
472 }
