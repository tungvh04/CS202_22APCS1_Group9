171sf::Texture CharacterState::resizeTexture(const sf::Texture& originalTexture, const sf::Vector2u& targetSize) {
172
173 sf::Image originalImage = originalTexture.copyToImage();
174 sf::Image resizedImage;
175 resizedImage.create(targetSize.x, targetSize.y, sf::Color::Transparent);
176
177 float xScale = static_cast<float>(targetSize.x) / originalImage.getSize().x;
178 float yScale = static_cast<float>(targetSize.y) / originalImage.getSize().y;
179
180 for (unsigned int x = 0; x < targetSize.x; ++x) {
181 for (unsigned int y = 0; y < targetSize.y; ++y) {
182 unsigned int origX = static_cast<unsigned int>(x / xScale);
183 unsigned int origY = static_cast<unsigned int>(y / yScale);
184
185 sf::Color pixelColor = originalImage.getPixel(origX, origY);
186 resizedImage.setPixel(x, y, pixelColor);
187 }
188 }
189
190 sf::Texture resizedTexture;
191 resizedTexture.loadFromImage(resizedImage);
192 resizedTexture.setSmooth(false);
193 return resizedTexture;
